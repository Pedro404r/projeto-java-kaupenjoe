import java.util.Scanner;

public class LoopExemplo {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Quantos números você quer digitar?");
        int n = scanner.nextInt();

        // --- Exemplo com while ---
        int somaWhile = 0; // soma para o loop while
        int i = 0;         // contador para while

        while (i < n) {
            System.out.println("Digite o número " + (i + 1) + " (while):");
            int numero = scanner.nextInt();
            somaWhile += numero;
            i++;
        }

        System.out.println("A soma dos números digitados com while é: " + somaWhile);

        // --- Exemplo com do-while ---
        int somaDoWhile = 0; // soma para o loop do-while
        int j = 0;           // contador para do-while

        if (n > 0) { // só roda o do-while se o número for maior que zero
            do {
                System.out.println("Digite o número " + (j + 1) + " (do-while):");
                int numero = scanner.nextInt();
                somaDoWhile += numero;
                j++;
            } while (j < n);

            System.out.println("A soma dos números digitados com do-while é: " + somaDoWhile);
        } else {
            System.out.println("Número de entradas deve ser maior que zero para o do-while.");
        }

        scanner.close();
    }
}

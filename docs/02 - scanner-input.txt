package whoami; // Pacote para organizar o código

import java.util.Scanner; // Importa a classe Scanner para entrada de dados pelo teclado

public class Main {
    public static void main(String[] args) {
        // Declaração e inicialização de variáveis (exemplo)
        int nomeExemplo = 1;    // Variável de exemplo (não utilizada para entrada de nome real)
        int idadeExemplo = 24;  // Variável de exemplo (não utilizada para entrada real)

        // Exibindo valores das variáveis exemplo no console
        System.out.println("Valor da variável nomeExemplo: " + nomeExemplo);
        System.out.println("Valor da variável idadeExemplo: " + idadeExemplo);

        // Criando um objeto Scanner para capturar entrada do usuário
        Scanner scanner = new Scanner(System.in);

        // Solicitando ao usuário que digite sua idade
        System.out.println("Por favor, insira sua idade:");

        // Lendo um número inteiro digitado pelo usuário e armazenando na variável 'idadeUsuario'
        int idadeUsuario = scanner.nextInt();

        // Exibindo a idade digitada
        System.out.println("Você escolheu a idade: " + idadeUsuario);

        // Solicitando ao usuário que digite seu nome
        System.out.println("Por favor, digite seu nome:");

        // Lendo o nome digitado pelo usuário (apenas uma palavra)
        String nomeUsuario = scanner.next();

        // Exibindo o nome digitado
        System.out.println("Seu nome é: " + nomeUsuario);

        // Mensagem final
        System.out.println("Formulário teste!");

        // Fechando o scanner para liberar recursos
        scanner.close();
    }
}

package whoami;

import java.util.Scanner;

public class Main {
    public static void main(String[] args) { // Declaração da main

       // Adição
       int a = 300; // variavel a que recebe 300
       int b = 400; // variavel b que recebe 400

       int peso = a + b; // soma (de exemplo coloquei peso)

       System.out.println("O seu peso é: " + peso); // 

       // Subtração
       int subt = a - b;
       System.out.println("O resultado da subtração é: " + subt);

       // Multiplicação
       int mult = a * b;
       System.out.println("O resultado da multiplicação é: " + mult);

       // Divisão
       int divi = a / b;
       System.out.println("O resultado da divisão é: " + divi);

       // Divisão com resto
       int divirst = b / a;
       System.out.println("O resultado da divisão com resto é: " + divirst);

       // Módulo
       int module = a % b;
       System.out.println("O resultado do módulo é: " + module);

 // Para usar outros recursos como valor absoluto: System.out.println("Teste" + Math.(abs por exemplo que seria absolute)(200);
    }
}
